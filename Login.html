    <script>

        function showCredentials() {

            var username = document.getElementById('username').value;

            var password = document.getElementById('password').value;


            // Store credentials in local storage

            localStorage.setItem('username', username);

            localStorage.setItem('password', password);


            // Hide login form and display a blank page

            document.getElementById('loginContainer').style.display = 'none';

            document.getElementById('result').style.display = 'none';

        }


        // Check if credentials are stored in local storage

        window.onload = function() {

            var storedUsername = localStorage.getItem('username');

            var storedPassword = localStorage.getItem('password');


            if (storedUsername && storedPassword) {

                document.getElementById('displayUsername').innerText = storedUsername;

                document.getElementById('displayPassword').innerText = storedPassword;

                document.getElementById('result').style.display = 'block';

            }

        }

    </script>
